#include <iostream>
using namespace std;

void swap(int* arr, int i, int j){
    
    int temp = arr[i];
    arr[i] = arr[j];
    arr[j] = temp;
}

int* selectionSort(int arr[], int n){
    
    for(int i = 0; i < n; i++){
        
        int min = i;
        for(int j = i + 1; j < n; j++){
            
            if(arr[j] < arr[min]){
                
                min = j;
            }
        }
        swap(arr, i, min);
    }
    
    return arr;
}

int* insertionSort(int arr[], int n){
    
    for(int i = 0; i < n; i++){
        int min = i;
        for(int j = i - 1; j >= 0; j--){
            
            if(arr[min] < arr[j]){
                
                swap(arr, j, min);
                min = j;
            }else
                break;
        }
    }
    
    return arr;
}

int main(){
    
    int arr[] = {-20, 10, 5, 1, 3, 0, 11, 12, -1, -2, -10};
    int n = sizeof(arr)/sizeof(*arr);
    
    //int *p = selectionSort(arr, n);
    int *p = insertionSort(arr, n);
    
    cout << n << endl;
    for(int i = 0; i < n; i++)
        cout << p[i] << " ";
    cout << endl;
    
    return 0;
}
